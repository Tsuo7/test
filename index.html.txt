    function createParticles(char) {
        const container = document.getElementById('bg-overlay');
        container.innerHTML = '';
        for(let i=0; i < 15; i++) {
            const p = document.createElement('div');
            p.className = 'particle';
            p.innerText = char;
            p.style.left = Math.random() * 100 + 'vw';
            p.style.top = Math.random() * 100 + 'vh';
            p.style.fontSize = (Math.random() * 20 + 10) + 'px';
            p.style.opacity = '0.3';
            container.appendChild(p);
            gsap.to(p, {
                y: (Math.random() - 0.5) * 300,
                x: (Math.random() - 0.5) * 300,
                duration: Math.random() * 4 + 3,
                repeat: -1, yoyo: true, ease: "sine.inOut"
            });
        }
    }

    function openAdmin() { document.getElementById('loginModal').style.display = 'flex'; }
    function closeModal(id) { document.getElementById(id).style.display = 'none'; }

    function handleLogin() {
        if(document.getElementById('user').value === 'TSUO' && document.getElementById('pass').value === '171088') {
            closeModal('loginModal');
            document.getElementById('adminPanel').style.display = 'block';
            loadAdminList();

            document.getElementById('myDiscord').value = owner.discord;
            document.getElementById('myBio').value = owner.bio;
        } else alert('ACESSO NEGADO');
    }

    function savePost() {
        const id = document.getElementById('editId').value;
        const data = {
            id: id ? parseInt(id) : Date.now(),
            title: document.getElementById('pTitle').value,
            btnName: document.getElementById('pBtnName').value || 'BAIXAR AGORA',
            btnLink: document.getElementById('pBtnLink').value,
            img: document.getElementById('pImg').value,
            desc: document.getElementById('pDesc').value,
            note: document.getElementById('pNote').value
        };

        if(!data.title) return alert('Insira um t√≠tulo!');

        if(id) {
            const idx = posts.findIndex(x => x.id === parseInt(id));
            posts[idx] = data;
        } else {
            posts.unshift(data);
        }

        localStorage.setItem('tsuo_final_posts', JSON.stringify(posts));
        alert('DADOS SINCRONIZADOS COM SUCESSO!');
        location.reload();
    }

    function renderFeed(search = "") {
        const feed = document.getElementById('postFeed');
        feed.innerHTML = '';
        const filtered = posts.filter(p => p.title.toLowerCase().includes(search.toLowerCase()));

        filtered.forEach(p => {
            const card = document.createElement('div');
            card.className = "neon-box p-6 group cursor-pointer hover:scale-105 transition-all duration-500 rounded-3xl overflow-hidden";
            card.innerHTML = `
                <div class="h-48 overflow-hidden mb-6 rounded-xl">
                    <img src="${p.img}" class="w-full h-full object-cover group-hover:scale-110 transition duration-700">
                </div>
                <h3 class="syncopate text-lg mb-4">${p.title}</h3>
                <div class="flex justify-between items-center mt-6 opacity-40 group-hover:opacity-100 transition">
                    <span class="text-[10px] syncopate">ABRIR ARQUIVO</span>
                    <i data-lucide="plus-circle" class="w-5 h-5"></i>
                </div>
            `;
            card.onclick = () => viewPost(p.id);
            feed.appendChild(card);
        });
        lucide.createIcons();
    }

    // --- ADMIN LISTA E PESQUISA ---
    function loadAdminList() {
        const list = document.getElementById('adminPostList');
        const s = document.getElementById('adminSearch').value.toLowerCase();
        list.innerHTML = '';
        posts.filter(p => p.title.toLowerCase().includes(s)).forEach(p => {
            const row = document.createElement('div');
            row.className = "flex justify-between items-center p-4 bg-white/5 rounded-lg border border-white/5";
            row.innerHTML = `
                <span class="text-xs font-bold">${p.title}</span>
                <div class="flex gap-4">
                    <button onclick="editPost(${p.id})" class="text-blue-400 text-[10px] syncopate">EDITAR</button>
                    <button onclick="duplicatePost(${p.id})" class="text-yellow-400 text-[10px] syncopate">COPIAR</button>
                    <button onclick="deletePost(${p.id})" class="text-red-500 text-[10px] syncopate">DELETAR</button>
                </div>
            `;
            list.appendChild(row);
        });
    }

    function editPost(id) {
        const p = posts.find(x => x.id === id);
        document.getElementById('editId').value = p.id;
        document.getElementById('pTitle').value = p.title;
        document.getElementById('pBtnName').value = p.btnName;
        document.getElementById('pBtnLink').value = p.btnLink;
        document.getElementById('pImg').value = p.img;
        document.getElementById('pDesc').value = p.desc;
        document.getElementById('pNote').value = p.note;
        document.getElementById('saveBtn').innerText = 'ATUALIZAR CONTE√öDO';
        gsap.to(".lg\\:col-span-2", { x: -10, repeat: 1, yoyo: true, duration: 0.1 });
    }

    function duplicatePost(id) {
        const p = {...posts.find(x => x.id === id)};
        p.id = Date.now();
        p.title += " (CLONE)";
        posts.unshift(p);
        localStorage.setItem('tsuo_final_posts', JSON.stringify(posts));
        loadAdminList();
        renderFeed();
    }

    function deletePost(id) {
        if(confirm('DELETAR PERMANENTEMENTE?')) {
            posts = posts.filter(x => x.id !== id);
            localStorage.setItem('tsuo_final_posts', JSON.stringify(posts));
            loadAdminList();
            renderFeed();
        }
    }
