    <script>
        // --- BASE DE DADOS LOCAL (LocalStorage) ---
        let posts = JSON.parse(localStorage.getItem('tsuo_data_posts')) || [];
        let config = JSON.parse(localStorage.getItem('tsuo_data_config')) || {
            title: "Tsuo Sociais",
            about: "Bem-vindo ao meu universo digital. Onde a inovação encontra a conveniência. Explore os meus projetos, scripts e muito mais.",
            social: { discord: '#', youtube: '#', instagram: '#', github: '#' }
        };

        // --- INICIALIZAÇÃO ---
        function init() {
            lucide.createIcons();
            renderSite();
            renderPosts();
            document.querySelectorAll('.animate-in').forEach((el, i) => {
                el.style.animationDelay = `${i * 0.1}s`;
            });

            // Conectar o color picker ao campo hidden
            document.getElementById('pColorPicker').addEventListener('input', (event) => {
                document.getElementById('pColor').value = event.target.value;
            });
            document.getElementById('pColorPicker').value = document.getElementById('pColor').value; // Inicia com a cor padrão
        }

        // --- SISTEMA DE LOGIN ---
        function toggleModal(id) { document.getElementById(id).classList.toggle('hidden'); }

        function handleLogin() {
            const u = document.getElementById('adminUser').value;
            const p = document.getElementById('adminPass').value;
            if(u === 'TSUO' && p === '171088') { // Credenciais atualizadas
                toggleModal('loginModal');
                document.getElementById('adminDashboard').classList.remove('hidden');
                renderAdminList();
                // Preencher campos de config no admin
                document.getElementById('sTitle').value = config.title;
                document.getElementById('sAbout').value = config.about;
                document.getElementById('sDiscord').value = config.social.discord;
                document.getElementById('sYoutube').value = config.social.youtube;
                document.getElementById('sInstagram').value = config.social.instagram;
                document.getElementById('sGithub').value = config.social.github;
            } else { alert("Acesso Negado: Usuário ou Senha incorretos."); }
        }

        function logout() { location.reload(); }

        // --- LOGICA DE POSTS ---
        function createPost() {
            const title = document.getElementById('pTitle').value;
            const desc = document.getElementById('pDesc').value;
            const img = document.getElementById('pImg').value || 'https://via.placeholder.com/400x250/331a40/e0e7ff?text=Tsuo+Content'; // Imagem padrão
            const theme = document.getElementById('pTheme').value;
            const color = document.getElementById('pColor').value;

            if(!title || !desc) return alert("Por favor, preencha o Título e a Descrição.");

            const newPost = { id: Date.now(), title, desc, img, theme, color };
            posts.unshift(newPost); // Novo post no topo
            saveAll();
            renderPosts();
            renderAdminList();
            alert("Conteúdo publicado com sucesso!");
            // Limpar campos
            document.getElementById('pTitle').value = '';
            document.getElementById('pDesc').value = '';
            document.getElementById('pImg').value = '';
            document.getElementById('pTheme').value = '';
            document.getElementById('pColor').value = '#8B5CF6';
            document.getElementById('pColorPicker').value = '#8B5CF6';
        }

        function deletePost(id) {
            if (confirm("Tem certeza que deseja remover este conteúdo?")) {
                posts = posts.filter(p => p.id !== id);
                saveAll();
                renderPosts();
                renderAdminList();
                alert("Conteúdo removido!");
            }
        }

        function updateSiteConfig() {
            config.title = document.getElementById('sTitle').value;
            config.about = document.getElementById('sAbout').value;
            config.social.discord = document.getElementById('sDiscord').value;
            config.social.youtube = document.getElementById('sYoutube').value;
            config.social.instagram = document.getElementById('sInstagram').value;
            config.social.github = document.getElementById('sGithub').value;
            saveAll();
            renderSite();
            alert("Página Inicial atualizada!");
        }

        function saveAll() {
            localStorage.setItem('tsuo_data_posts', JSON.stringify(posts));
            localStorage.setItem('tsuo_data_config', JSON.stringify(config));
        }

        // --- RENDERIZAÇÃO ---
        function renderSite() {
            document.getElementById('displayTitle').innerText = config.title;
            document.getEle
