<script>
    // --- AUDIO SYSTEM ---
    const sfx = {
        click: new Howl({ src: ['https://assets.mixkit.co/sfx/preview/mixkit-modern-technology-select-3124.mp3'], volume: 0.3 }),
        bgm: new Howl({ src: ['https://www.soundhelix.com/examples/mp3/SoundHelix-Song-15.mp3'], loop: true, volume: 0.1 })
    };

    let musicOn = false;
    function toggleMusic() {
        if(!musicOn) sfx.bgm.play(); else sfx.bgm.pause();
        musicOn = !musicOn;
        document.getElementById('musicIcon').style.color = musicOn ? '#bc13fe' : '#fff';
    }

    // --- THREE.JS BACKGROUND ---
    const scene = new THREE.Scene();
    const camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 0.1, 1000);
    const renderer = new THREE.WebGLRenderer({ alpha: true });
    renderer.setSize(window.innerWidth, window.innerHeight);
    document.getElementById('canvas-container').appendChild(renderer.domElement);

    const starsGeo = new THREE.BufferGeometry();
    const starsCoords = [];
    for(let i=0; i<6000; i++) starsCoords.push(THREE.MathUtils.randFloatSpread(2000), THREE.MathUtils.randFloatSpread(2000), THREE.MathUtils.randFloatSpread(2000));
    starsGeo.setAttribute('position', new THREE.Float32BufferAttribute(starsCoords, 3));
    const stars = new THREE.Points(starsGeo, new THREE.PointsMaterial({ color: 0xbc13fe, size: 1.5 }));
    scene.add(stars);
    camera.position.z = 1;

    function animateBg() {
        requestAnimationFrame(animateBg);
        stars.rotation.y += 0.0005;
        renderer.render(scene, camera);
    }
    animateBg();

    // --- CORE LOGIC ---
    let posts = JSON.parse(localStorage.getItem('tsuo_v4_posts')) || [];

    function openLogin() { sfx.click.play(); document.getElementById('loginModal').style.display = 'flex'; }
    function closeModal(id) { document.getElementById(id).style.display = 'none'; }
    
    function validateLogin() {
        if(document.getElementById('user').value === 'TSUO' && document.getElementById('pass').value === '171088') {
            closeModal('loginModal');
            document.getElementById('adminPanel').style.display = 'block';
            renderAdmin();
        } else alert('ERROR: UNAUTHORIZED');
    }

    function publishPost() {
        const p = {
            id: Date.now(),
            title: document.getElementById('pTitle').value,
            cat: document.getElementById('pCat').value,
            img: document.getElementById('pImg').value,
            desc: document.getElementById('pDesc').value
        };
        posts.unshift(p);
        localStorage.setItem('tsuo_v4_posts', JSON.stringify(posts));
        renderFeed();
        renderAdmin();
        alert('RECORD_STORED');
    }

    function renderFeed(filter = "", cat = "") {
        const feed = document.getElementById('postFeed');
        let filtered = posts;
        if(filter) filtered = posts.filter(x => x.title.toLowerCase().includes(filter.toLowerCase()));
        if(cat) filtered = posts.filter(x => x.cat === cat);

        feed.innerHTML = filtered.map(p => `
            <div class="glass p-6 group cursor-pointer hover:border-purple-500 transition-all duration-500" onclick="viewPost(${p.id})">
                <div class="h-40 overflow-hidden mb-4 rounded-xl border border-white/5">
                    <img src="${p.img}" class="w-full h-full object-cover group-hover:scale-110 transition duration-700" onerror="this.src='https://via.placeholder.com/400x200?text=IMAGE_ERROR'">
                </div>
                <span class="category-tag tag-${p.cat}">${p.cat}</span>
                <h4 class="orbitron text-lg font-bold mt-3">${p.title}</h4>
                <p class="text-[10px] opacity-40 mt-4 orbitron">CLICK_TO_EXPAND_DATABASE_FILE</p>
            </div>
        `).join('');
    }

    function viewPost(id) {
        const p = posts.find(x => x.id === id);
        const postPage = document.getElementById('postPage');
        const content = document.getElementById('postContent');
        
        content.innerHTML = `
            <div class="flex flex-col md:flex-row gap-10">
                <div class="md:w-1/2">
                    <img src="${p.img}" class="w-full rounded-2xl shadow-2xl border border-purple-500/30">
                </div>
                <div class="md:w-1/2">
                    <span class="category-tag tag-${p.cat} mb-4 inline-block">${p.cat}</span>
                    <h2 class="orbitron text-4xl font-black mb-6 neon-text">${p.title}</h2>
                    <div class="text-gray-300 leading-relaxed mb-10">${p.desc}</div>
                    <button class="bg-purple-600 px-10 py-4 orbitron font-bold hover:bg-purple-500 transition rounded-full">DOWNLOAD_NOW</button>
                </div>
            </div>
        `;
        postPage.style.display = 'block';
        gsap.from("#postContent", { y: 100, opacity: 0, duration: 1 });
    }

    function closePost() { document.getElementById('postPage').style.display = 'none'; }

    // --- SNAKE GAME LOGIC ---
    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');
    let snake = [{x: 200, y: 200}], food = {x: 100, y: 100}, dx = 20, dy = 0;

    function gameLoop() {
        if(document.getElementById('gameView').classList.contains('hidden')) return;
        setTimeout(() => {
            clearCanvas();
            drawFood();
            advanceSnake();
            drawSnake();
            gameLoop();
        }, 100);
    }

    function clearCanvas() { ctx.fillStyle = "black"; ctx.fillRect(0,0,400,400); }
    function drawSnake() { ctx.fillStyle = "#bc13fe"; snake.forEach(s => ctx.fillRect(s.x, s.y, 18, 18)); }
    function advanceSnake() {
        const head = {x: snake[0].x + dx, y: snake[0].y + dy};
        snake.unshift(head);
        if(head.x === food.x && head.y === food.y) generateFood(); else snake.pop();
        if(head.x<0 || head.x>=400 || head.y<0 || head.y>=400) resetGame();
    }
    function drawFood() { ctx.fillStyle = "#00f2ff"; ctx.fillRect(food.x, food.y, 18, 18); }
    function generateFood() {
        food.x = Math.floor(Math.random() * 20) * 20;
        food.y = Math.floor(Math.random() * 20) * 20;
    }
    function resetGame() { snake = [{x: 200, y: 200}]; dx = 20; dy = 0; }

    window.addEventListener('keydown', e => {
        if(e.key === 'ArrowUp' && dy === 0) { dx = 0; dy = -20; }
        if(e.key === 'ArrowDown' && dy === 0) { dx = 0; dy = 20; }
        if(e.key === 'ArrowLeft' && dx === 0) { dx = -20; dy = 0; }
        if(e.key === 'ArrowRight' && dx === 0) { dx = 20; dy = 0; }
    });

    function openTab(t) { 
        if(t === 'game') { document.getElementById('gameView').classList.remove('hidden'); gameLoop(); }
    }
    function closeTab(t) { document.getElementById('gameView').classList.add('hidden'); }

    // --- MAGNÉTICO & SEARCH ---
    document.getElementById('searchInput').oninput = (e) => renderFeed(e.target.value);
    function filterCat(c) { renderFeed("", c); }

    window.onload = () => {
        lucide.createIcons();
        renderFeed();
        // Animação de título magnético
        const title = document.getElementById('magTitle');
        title.innerHTML = title.innerText.split('').map(l => `<span class="mag-text">${l}</span>`).join('');
    };
    
    function renderAdmin() {
        const list = document.getElementById('adminList');
        list.innerHTML = posts.map(p => `
            <div class="flex justify-between bg-white/5 p-3 text-[10px] orbitron border border-white/5">
                <span>${p.title}</span>
                <button onclick="deletePost(${p.id})" class="text-red-500">DELETE</button>
            </div>
        `).join('');
    }
    function deletePost(id) { posts = posts.filter(x => x.id !== id); localStorage.setItem('tsuo_v4_posts', JSON.stringify(posts)); renderAdmin(); renderFeed(); }
</script>
